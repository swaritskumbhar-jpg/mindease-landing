/* screens/ChatScreen.tsx (excerpt) */
import { useState } from 'react';
import { GiftedChat } from 'react-native-gifted-chat';
import { Configuration, OpenAIApi } from 'openai';

const openai = new OpenAIApi(new Configuration({ apiKey: process.env.EXPO_PUBLIC_OPENAI_KEY }));

async function getAIResponse(message: string) {
  const systemPrompt = `You are MindEase, a warm, friendly companion for students. 
  Be encouraging, give practical study/breathing tips, never give medical advice. 
  Always end with empathy and remind users you are not a therapist.`;
  
  const res = await openai.createChatCompletion({
    model: "gpt-4o-mini",
    messages: [{ role: "system", content: systemPrompt }, { role: "user", content: message }]
  });
  return res.data.choices[0].message.content;
}
